\[ \max Z = 3x_1 + 4x_2 + x_3 + 7x_4\]

{\centering
  subject to
  \vspace{2mm}

  \sysdelim..%
  \sysalign{r,r}%
  \systeme[x_1x_2x_3x_4]%

    8x_1 + 3x_2 + 4x_3 + x_4 \leq 7,
    2x_1 + 6x_2 + x_3 + 5x_4 \leq 3,
